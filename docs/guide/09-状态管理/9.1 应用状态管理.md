# 第九章：状态管理——从Vuex/Pinia到MVVM

在大型应用中，状态管理是核心问题。Vue生态中的Vuex和Pinia提供了集中式状态管理方案。在Avalonia中，我们使用MVVM模式结合依赖注入（DI）来实现类似的功能。

## 9.1 应用状态管理

### 9.1.1 Vuex/Pinia vs Service + DI

**Pinia Store（Vue 3）：**

```typescript
// stores/user.ts
import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', {
  state: () => ({
    currentUser: null as User | null,
    isAuthenticated: false,
    token: null as string | null
  }),

  getters: {
    userName: (state) => state.currentUser?.name || '游客',
    isAdmin: (state) => state.currentUser?.role === 'admin'
  },

  actions: {
    async login(credentials: Credentials) {
      const response = await api.login(credentials)
      this.currentUser = response.user
      this.token = response.token
      this.isAuthenticated = true
    },

    logout() {
      this.currentUser = null
      this.token = null
      this.isAuthenticated = false
    }
  }
})
```

**在组件中使用：**

```vue
<script setup>
import { useUserStore } from '@/stores/user'

const userStore = useUserStore()

// 访问状态
console.log(userStore.currentUser)
console.log(userStore.userName)  // getter

// 调用action
userStore.login({ username: 'alice', password: '123456' })
</script>
```

**Avalonia实现（Service + DI）：**

```csharp
// Services/UserService.cs
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;

namespace MyApp.Services
{
    // 状态服务（类似Pinia Store）
    public partial class UserService : ObservableObject
    {
        // State
        [ObservableProperty]
        private User? currentUser = null;

        [ObservableProperty]
        private bool isAuthenticated = false;

        [ObservableProperty]
        private string? token = null;

        // Getters（计算属性）
        public string UserName => CurrentUser?.Name ?? "游客";
        public bool IsAdmin => CurrentUser?.Role == "admin";

        partial void OnCurrentUserChanged(User? value)
        {
            OnPropertyChanged(nameof(UserName));
            OnPropertyChanged(nameof(IsAdmin));
        }

        // Actions（方法）
        public async Task LoginAsync(Credentials credentials)
        {
            var response = await api.LoginAsync(credentials);
            CurrentUser = response.User;
            Token = response.Token;
            IsAuthenticated = true;
        }

        public void Logout()
        {
            CurrentUser = null;
            Token = null;
            IsAuthenticated = false;
        }
    }

    // 数据模型
    public class User
    {
        public string Name { get; set; } = string.Empty;
        public string Role { get; set; } = string.Empty;
    }

    public class Credentials
    {
        public string Username { get; set; } = string.Empty;
        public string Password { get; set; } = string.Empty;
    }
}
```

**注册服务（类似Pinia的createPinia）：**

```csharp
// App.axaml.cs
using Microsoft.Extensions.DependencyInjection;
using MyApp.Services;

public class App : Application
{
    public override void Initialize()
    {
        // 配置依赖注入
        var services = new ServiceCollection();

        // 注册服务（单例）
        services.AddSingleton<UserService>();  // 全局状态
        services.AddSingleton<ThemeService>();
        services.AddTransient<MainViewModel>();

        var provider = services.BuildServiceProvider();

        // 保存到资源中
        Resources["ServiceProvider"] = provider;

        base.Initialize();
    }
}
```

**在ViewModel中使用：**

```csharp
// ViewModels/MainViewModel.cs
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using MyApp.Services;

public partial class MainViewModel : ObservableObject
{
    private readonly UserService userService;

    // 访问全局状态
    public User? CurrentUser => userService.CurrentUser;
    public string UserName => userService.UserName;
    public bool IsAuthenticated => userService.IsAuthenticated;

    public MainViewModel(UserService userService)
    {
        this.userService = userService;

        // 监听全局状态变化（类似store.$subscribe）
        userService.PropertyChanged += (s, e) =>
        {
            OnPropertyChanged(e.PropertyName);
        };
    }

    [RelayCommand]
    private async Task Login()
    {
        await userService.LoginAsync(new Credentials
        {
            Username = "alice",
            Password = "123456"
        });
    }

    [RelayCommand]
    private void Logout() => userService.Logout();
}
```

### 9.1.2 完整示例：购物车状态管理

**Pinia实现：**

```typescript
// stores/cart.ts
export const useCartStore = defineStore('cart', {
  state: () => ({
    items: [] as CartItem[],
    checkoutStatus: null as string | null
  }),

  getters: {
    totalCount: (state) =>
      state.items.reduce((sum, item) => sum + item.quantity, 0),

    totalPrice: (state) =>
      state.items.reduce((sum, item) => sum + item.price * item.quantity, 0),

    isEmpty: (state) => state.items.length === 0
  },

  actions: {
    addItem(product: Product) {
      const existingItem = this.items.find(item => item.id === product.id)
      if (existingItem) {
        existingItem.quantity++
      } else {
        this.items.push({ ...product, quantity: 1 })
      }
    },

    removeItem(productId: number) {
      const index = this.items.findIndex(item => item.id === productId)
      if (index > -1) {
        this.items.splice(index, 1)
      }
    },

    updateQuantity(productId: number, quantity: number) {
      const item = this.items.find(item => item.id === productId)
      if (item) {
        item.quantity = quantity
      }
    },

    clearCart() {
      this.items = []
    },

    async checkout() {
      this.checkoutStatus = 'pending'
      try {
        await api.checkout(this.items)
        this.items = []
        this.checkoutStatus = 'successful'
      } catch (error) {
        this.checkoutStatus = 'failed'
      }
    }
  }
})
```

**Avalonia实现：**

```csharp
// Services/CartService.cs
using CommunityToolkit.Mvvm.ComponentModel;
using System.Collections.ObjectModel;

namespace MyApp.Services
{
    public partial class CartService : ObservableObject
    {
        // State
        public ObservableCollection<CartItem> Items { get; }

        [ObservableProperty]
        private string? checkoutStatus = null;

        // Getters（计算属性）
        public int TotalCount => Items.Sum(item => item.Quantity);
        public decimal TotalPrice => Items.Sum(item => item.Price * item.Quantity);
        public bool IsEmpty => Items.Count == 0;

        // Events
        public event EventHandler<CartItemEventArgs>? ItemAdded;
        public event EventHandler<int>? ItemRemoved;

        public CartService()
        {
            Items = new ObservableCollection<CartItem>();
            Items.CollectionChanged += (s, e) =>
            {
                OnPropertyChanged(nameof(TotalCount));
                OnPropertyChanged(nameof(TotalPrice));
                OnPropertyChanged(nameof(IsEmpty));
            };
        }

        // Actions
        public void AddItem(Product product)
        {
            var existingItem = Items.FirstOrDefault(item => item.ProductId == product.Id);
            if (existingItem != null)
            {
                existingItem.Quantity++;
                UpdateTotals();
            }
            else
            {
                var cartItem = new CartItem
                {
                    ProductId = product.Id,
                    ProductName = product.Name,
                    Price = product.Price,
                    Quantity = 1
                };
                Items.Add(cartItem);
            }
            ItemAdded?.Invoke(this, new CartItemEventArgs(existingItem ?? Items.Last()));
        }

        public void RemoveItem(int productId)
        {
            var item = Items.FirstOrDefault(i => i.ProductId == productId);
            if (item != null)
            {
                Items.Remove(item);
                ItemRemoved?.Invoke(this, productId);
            }
        }

        public void UpdateQuantity(int productId, int quantity)
        {
            var item = Items.FirstOrDefault(i => i.ProductId == productId);
            if (item != null)
            {
                if (quantity <= 0) RemoveItem(productId);
                else { item.Quantity = quantity; UpdateTotals(); }
            }
        }

        public void ClearCart() => Items.Clear();

        public async Task CheckoutAsync()
        {
            CheckoutStatus = "pending";
            try
            {
                await api.CheckoutAsync(Items.ToList());
                Items.Clear();
                CheckoutStatus = "successful";
            }
            catch (Exception)
            {
                CheckoutStatus = "failed";
            }
        }

        private void UpdateTotals()
        {
            OnPropertyChanged(nameof(TotalCount));
            OnPropertyChanged(nameof(TotalPrice));
        }
    }

    // CartItem
    public partial class CartItem : ObservableObject
    {
        public int ProductId { get; set; }
        public string ProductName { get; set; } = string.Empty;
        public decimal Price { get; set; }

        [ObservableProperty]
        private int quantity = 0;

        public decimal Subtotal => Price * Quantity;
    }

    public class Product
    {
        public int Id { get; set; }
        public string Name { get; set; } = string.Empty;
        public decimal Price { get; set; }
    }

    public class CartItemEventArgs : EventArgs
    {
        public CartItem Item { get; }

        public CartItemEventArgs(CartItem item)
        {
            Item = item;
        }
    }
}
```

// ViewModels/CartViewModel.cs
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;

public partial class CartViewModel : ObservableObject
{
    private readonly CartService cartService;

    public ObservableCollection<CartItem> Items => cartService.Items;
    public int TotalCount => cartService.TotalCount;
    public decimal TotalPrice => cartService.TotalPrice;
    public bool IsEmpty => cartService.IsEmpty;

    public CartViewModel(CartService cartService)
    {
        this.cartService = cartService;

        // 监听状态变化
        cartService.PropertyChanged += (s, e) =>
        {
            OnPropertyChanged(e.PropertyName);
        };
    }

    [RelayCommand]
    private async Task Checkout() => await cartService.CheckoutAsync();

    [RelayCommand]
    private void Remove(int productId) => cartService.RemoveItem(productId);
}
