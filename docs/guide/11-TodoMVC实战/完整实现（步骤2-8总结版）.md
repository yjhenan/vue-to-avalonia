### TodoService完整实现

```csharp
// Services/TodoService.cs
using System.Collections.ObjectModel;
using System.IO;
using System.Linq;
using System.Text.Json;
using System.Threading.Tasks;
using TodoMVC.Models;

namespace TodoMVC.Services
{
    public class TodoService : ITodoService
    {
        private const string DataFile = "todos.json";
        public ObservableCollection<TodoItem> Todos { get; }

        public TodoService()
        {
            Todos = new ObservableCollection<TodoItem>();
        }

        public async Task LoadAsync()
        {
            if (File.Exists(DataFile))
            {
                var json = await File.ReadAllTextAsync(DataFile);
                var items = JsonSerializer.Deserialize<TodoItem[]>(json);
                if (items != null)
                {
                    Todos.Clear();
                    foreach (var item in items)
                        Todos.Add(item);
                }
            }
        }

        public async Task SaveAsync()
        {
            var json = JsonSerializer.Serialize(Todos.ToArray(), new JsonSerializerOptions
            {
                WriteIndented = true
            });
            await File.WriteAllTextAsync(DataFile, json);
        }

        public void Add(string title)
        {
            if (string.IsNullOrWhiteSpace(title)) return;

            Todos.Add(new TodoItem
            {
                Id = DateTime.Now.Ticks,
                Title = title.Trim(),
                Completed = false
            });
            SaveAsync().ConfigureAwait(false);
        }

        public void Remove(long id)
        {
            var item = Todos.FirstOrDefault(t => t.Id == id);
            if (item != null)
            {
                Todos.Remove(item);
                SaveAsync().ConfigureAwait(false);
            }
        }

        public void Update(TodoItem item)
        {
            SaveAsync().ConfigureAwait(false);
        }

        public void ClearCompleted()
        {
            var completed = Todos.Where(t => t.Completed).ToList();
            foreach (var item in completed)
                Todos.Remove(item);
            SaveAsync().ConfigureAwait(false);
        }
    }
}
```

### MainViewModel完整实现

```csharp
// ViewModels/MainViewModel.cs
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using System;
using System.Collections.ObjectModel;
using System.Linq;
using TodoMVC.Models;
using TodoMVC.Services;

namespace TodoMVC.ViewModels
{
    public partial class MainViewModel : ViewModelBase
    {
        private readonly TodoService todoService;

        [ObservableProperty]
        private string newTodoTitle = string.Empty;

        [ObservableProperty]
        private string filter = "all";

        partial void OnFilterChanged(string value) => OnPropertyChanged(nameof(FilteredTodos));

        public ObservableCollection<TodoItem> Todos => todoService.Todos;

        public ObservableCollection<TodoItem> FilteredTodos
        {
            get
            {
                var filtered = Filter switch
                {
                    "active" => Todos.Where(t => !t.Completed),
                    "completed" => Todos.Where(t => t.Completed),
                    _ => Todos
                };
                return new ObservableCollection<TodoItem>(filtered);
            }
        }

        public int RemainingCount => Todos.Count(t => !t.Completed);
        public int CompletedCount => Todos.Count(t => t.Completed);
        public bool AllDone
        {
            get => Todos.Count > 0 && Todos.All(t => t.Completed);
            set
            {
                foreach (var todo in Todos)
                    todo.Completed = value;
                todoService.SaveAsync().ConfigureAwait(false);
            }
        }

        public MainViewModel(TodoService todoService)
        {
            this.todoService = todoService;
            Todos.CollectionChanged += (s, e) => UpdateCounts();
            todoService.LoadAsync().ConfigureAwait(false);
        }

        [RelayCommand]
        private void AddTodo()
        {
            if (string.IsNullOrWhiteSpace(NewTodoTitle)) return;
            todoService.Add(NewTodoTitle);
            NewTodoTitle = string.Empty;
            UpdateCounts();
        }

        [RelayCommand]
        private void RemoveTodo(long id)
        {
            todoService.Remove(id);
            UpdateCounts();
        }

        [RelayCommand]
        private void SetFilter(string newFilter) => Filter = newFilter;

        [RelayCommand]
        private void ClearCompleted()
        {
            todoService.ClearCompleted();
            UpdateCounts();
        }

        private void UpdateCounts()
        {
            OnPropertyChanged(nameof(RemainingCount));
            OnPropertyChanged(nameof(CompletedCount));
            OnPropertyChanged(nameof(AllDone));
            OnPropertyChanged(nameof(FilteredTodos));
        }
    }
}
```

### 启动和依赖注入

```csharp
// App.axaml.cs
using Avalonia;
using Avalonia.Markup.Xaml;
using Microsoft.Extensions.DependencyInjection;
using TodoMVC.Services;
using TodoMVC.ViewModels;
using TodoMVC.Views;

namespace TodoMVC
{
    public class App : Application
    {
        public override void Initialize()
        {
            AvaloniaXamlLoader.Load(this);

            // 配置DI
            var services = new ServiceCollection();
            services.AddSingleton<TodoService>();
            services.AddTransient<MainViewModel>();

            Resources["ServiceProvider"] = services.BuildServiceProvider();
        }

        public override void OnFrameworkInitializationCompleted()
        {
            var provider = (ServiceProvider)Resources["ServiceProvider"]!;
            var viewModel = provider.GetRequiredService<MainViewModel>();

            if (ApplicationLifetime is IClassicDesktopStyleApplicationLifetime desktop)
            {
                desktop.MainWindow = new MainWindow
                {
                    DataContext = viewModel
                };
            }

            base.OnFrameworkInitializationCompleted();
        }
    }
}
```
