## 11.1 Vue TodoMVC实现（回顾）

先快速回顾Vue的实现，作为对比基准。

**Store（Pinia）：**

```typescript
// stores/todo.ts
import { defineStore } from 'pinia'

export const useTodoStore = defineStore('todo', {
  state: () => ({
    todos: JSON.parse(localStorage.getItem('todos') || '[]') as Todo[],
    filter: 'all' as FilterType
  }),

  getters: {
    filteredTodos: (state) => {
      switch (state.filter) {
        case 'active':
          return state.todos.filter(t => !t.completed)
        case 'completed':
          return state.todos.filter(t => t.completed)
        default:
          return state.todos
      }
    },

    remaining: (state) =>
      state.todos.filter(t => !t.completed).length,

    allDone: (state) =>
      state.todos.length > 0 && state.todos.every(t => t.completed)
  },

  actions: {
    addTodo(title: string) {
      if (title.trim()) {
        this.todos.push({
          id: Date.now(),
          title: title.trim(),
          completed: false
        })
        this.saveTodos()
      }
    },

    removeTodo(id: number) {
      const index = this.todos.findIndex(t => t.id === id)
      if (index > -1) {
        this.todos.splice(index, 1)
        this.saveTodos()
      }
    },

    updateTodo(todo: Todo) {
      const index = this.todos.findIndex(t => t.id === todo.id)
      if (index > -1) {
        this.todos[index] = todo
        this.saveTodos()
      }
    },

    toggleAll(completed: boolean) {
      this.todos.forEach(t => t.completed = completed)
      this.saveTodos()
    },

    clearCompleted() {
      this.todos = this.todos.filter(t => !t.completed)
      this.saveTodos()
    },

    setFilter(filter: FilterType) {
      this.filter = filter
    },

    saveTodos() {
      localStorage.setItem('todos', JSON.stringify(this.todos))
    }
  }
})

type FilterType = 'all' | 'active' | 'completed'

interface Todo {
  id: number
  title: string
  completed: boolean
}
```

**组件：**

```vue
<!-- App.vue -->
<template>
  <div class="todo-app">
    <header class="header">
      <h1>todos</h1>
      <input
        v-model="newTodo"
        @keyup.enter="addTodo"
        class="new-todo"
        placeholder="What needs to be done?"
        autofocus
      />
    </header>

    <section class="main" v-show="todos.length">
      <input
        id="toggle-all"
        type="checkbox"
        class="toggle-all"
        v-model="allDone"
      />
      <label for="toggle-all">Mark all as complete</label>

      <ul class="todo-list">
        <li
          v-for="todo in filteredTodos"
          :key="todo.id"
          :class="{ completed: todo.completed, editing: todo.id === editingId }"
          class="todo"
        >
          <div class="view">
            <input
              type="checkbox"
              class="toggle"
              v-model="todo.completed"
              @change="updateTodo(todo)"
            />
            <label @dblclick="startEditing(todo)">{{ todo.title }}</label>
            <button class="destroy" @click="removeTodo(todo.id)"></button>
          </div>

          <input
            v-if="editingId === todo.id"
            v-model="todo.title"
            @keyup.enter="finishEditing(todo)"
            @keyup.escape="cancelEditing"
            @blur="finishEditing(todo)"
            class="edit"
            v-focus
          />
        </li>
      </ul>
    </section>

    <footer class="footer" v-show="todos.length">
      <span class="todo-count">
        <strong>{{ remaining }}</strong> item{{ remaining !== 1 ? 's' : '' }} left
      </span>

      <ul class="filters">
        <li>
          <a
            href="#"
            :class="{ selected: filter === 'all' }"
            @click="setFilter('all')"
          >All</a>
        </li>
        <li>
          <a
            href="#"
            :class="{ selected: filter === 'active' }"
            @click="setFilter('active')"
          >Active</a>
        </li>
        <li>
          <a
            href="#"
            :class="{ selected: filter === 'completed' }"
            @click="setFilter('completed')"
          >Completed</a>
        </li>
      </ul>

      <button
        v-show="todos.length > remaining"
        class="clear-completed"
        @click="clearCompleted"
      >
        Clear completed
      </button>
    </footer>
  </div>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { useTodoStore } from '@/stores/todo'
import { storeToRefs } from 'pinia'

const todoStore = useTodoStore()
const { todos, filteredTodos, remaining, allDone, filter } = storeToRefs(todoStore)

const newTodo = ref('')
const editingId = ref<number | null>(null)

function addTodo() {
  todoStore.addTodo(newTodo.value)
  newTodo.value = ''
}

function removeTodo(id: number) {
  todoStore.removeTodo(id)
}

function updateTodo(todo: any) {
  todoStore.updateTodo(todo)
}

function startEditing(todo: any) {
  editingId.value = todo.id
}

function finishEditing(todo: any) {
  if (todo.title.trim()) {
    updateTodo(todo)
  } else {
    removeTodo(todo.id)
  }
  editingId.value = null
}

function cancelEditing() {
  editingId.value = null
}

function setFilter(filter: 'all' | 'active' | 'completed') {
  todoStore.setFilter(filter)
}

function clearCompleted() {
  todoStore.clearCompleted()
}

// 自定义指令：自动聚焦
const vFocus = {
  mounted(el: HTMLInputElement) {
    el.focus()
  }
}
</script>
```
