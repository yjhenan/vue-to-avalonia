## 4.5 实战代码对比（详细）

### 4.5.1 案例一：表单双向绑定

**需求：**用户注册表单，包含用户名、邮箱、密码，实时验证。

**Vue实现：**

```vue
<template>
  <div class="register-form">
    <h2>用户注册</h2>

    <form @submit.prevent="handleSubmit">
      <!-- 用户名 -->
      <div class="form-group">
        <label>用户名</label>
        <input
          v-model="form.username"
          @blur="validateUsername"
          placeholder="请输入用户名"
        />
        <span v-if="errors.username" class="error">
          {{ errors.username }}
        </span>
      </div>

      <!-- 邮箱 -->
      <div class="form-group">
        <label>邮箱</label>
        <input
          v-model="form.email"
          type="email"
          @blur="validateEmail"
          placeholder="请输入邮箱"
        />
        <span v-if="errors.email" class="error">
          {{ errors.email }}
        </span>
      </div>

      <!-- 密码 -->
      <div class="form-group">
        <label>密码</label>
        <input
          v-model="form.password"
          type="password"
          @blur="validatePassword"
          placeholder="请输入密码"
        />
        <span v-if="errors.password" class="error">
          {{ errors.password }}
        </span>
      </div>

      <button type="submit" :disabled="!isFormValid">
        注册
      </button>
    </form>
  </div>
</template>

<script setup>
import { reactive, computed, ref } from 'vue'

const form = reactive({
  username: '',
  email: '',
  password: ''
})

const errors = reactive({
  username: '',
  email: '',
  password: ''
})

// 验证方法
function validateUsername() {
  if (!form.username) {
    errors.username = '用户名不能为空'
  } else if (form.username.length < 3) {
    errors.username = '用户名至少3个字符'
  } else {
    errors.username = ''
  }
}

function validateEmail() {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
  if (!form.email) {
    errors.email = '邮箱不能为空'
  } else if (!emailRegex.test(form.email)) {
    errors.email = '邮箱格式不正确'
  } else {
    errors.email = ''
  }
}

function validatePassword() {
  if (!form.password) {
    errors.password = '密码不能为空'
  } else if (form.password.length < 6) {
    errors.password = '密码至少6个字符'
  } else {
    errors.password = ''
  }
}

// 计算属性：表单是否有效
const isFormValid = computed(() => {
  return form.username &&
         form.email &&
         form.password &&
         !errors.username &&
         !errors.email &&
         !errors.password
})

// 提交表单
function handleSubmit() {
  validateUsername()
  validateEmail()
  validatePassword()

  if (isFormValid.value) {
    console.log('提交表单：', form)
  }
}
</script>

<style scoped>
.error {
  color: red;
  font-size: 12px;
}

button:disabled {
  opacity: 0.5;
  cursor: not-allowed;
}
</style>
```

**RegisterViewModel.cs：**
```csharp
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using System.Text.RegularExpressions;

public partial class RegisterViewModel : ObservableObject
{
    // ===== 属性 =====
    [ObservableProperty]
    private string username = string.Empty;

    [ObservableProperty]
    private string email = string.Empty;

    [ObservableProperty]
    private string password = string.Empty;

    [ObservableProperty]
    private string usernameError = string.Empty;

    [ObservableProperty]
    private string emailError = string.Empty;

    [ObservableProperty]
    private string passwordError = string.Empty;

    // ===== 计算属性 =====
    public bool HasUsernameError => !string.IsNullOrEmpty(UsernameError);
    public bool HasEmailError => !string.IsNullOrEmpty(EmailError);
    public bool HasPasswordError => !string.IsNullOrEmpty(PasswordError);

    public bool IsFormValid =>
        !string.IsNullOrEmpty(Username) &&
        !string.IsNullOrEmpty(Email) &&
        !string.IsNullOrEmpty(Password) &&
        !HasUsernameError &&
        !HasEmailError &&
        !HasPasswordError;

    // ===== 属性变更回调 =====
    partial void OnUsernameChanged(string value)
    {
        ValidateUsername();
        OnPropertyChanged(nameof(IsFormValid));
    }

    partial void OnEmailChanged(string value)
    {
        ValidateEmail();
        OnPropertyChanged(nameof(IsFormValid));
    }

    partial void OnPasswordChanged(string value)
    {
        ValidatePassword();
        OnPropertyChanged(nameof(IsFormValid));
    }

    // ===== Commands =====
    [RelayCommand]
    private void Submit()
    {
        if (IsFormValid)
        {
            System.Diagnostics.Debug.WriteLine($"提交表单：{Username}, {Email}");
            // 实际提交逻辑
        }
    }

    // ===== 验证方法 =====
    private void ValidateUsername()
    {
        if (string.IsNullOrEmpty(Username))
        {
            UsernameError = "用户名不能为空";
        }
        else if (Username.Length < 3)
        {
            UsernameError = "用户名至少3个字符";
        }
        else
        {
            UsernameError = string.Empty;
        }
        OnPropertyChanged(nameof(HasUsernameError));
    }

    private void ValidateEmail()
    {
        var emailRegex = new Regex(@"^[^\s@]+@[^\s@]+\.[^\s@]+$");

        if (string.IsNullOrEmpty(Email))
        {
            EmailError = "邮箱不能为空";
        }
        else if (!emailRegex.IsMatch(Email))
        {
            EmailError = "邮箱格式不正确";
        }
        else
        {
            EmailError = string.Empty;
        }
        OnPropertyChanged(nameof(HasEmailError));
    }

    private void ValidatePassword()
    {
        if (string.IsNullOrEmpty(Password))
        {
            PasswordError = "密码不能为空";
        }
        else if (Password.Length < 6)
        {
            PasswordError = "密码至少6个字符";
        }
        else
        {
            PasswordError = string.Empty;
        }
        OnPropertyChanged(nameof(HasPasswordError));
    }
}
```

**对比总结：**

| 方面 | Vue | Avalonia |
|-----|-----|----------|
| **表单绑定** | `v-model` | `{Binding ..., Mode=TwoWay}` |
| **验证触发** | `@blur` | 属性setter中调用 |
| **错误显示** | `v-if` | `IsVisible`绑定 |
| **表单状态** | `computed` | 计算属性 + 手动通知 |
| **提交逻辑** | `@submit` | `Command` |
| **代码量** | 约80行 | 约120行 + XAML |

### 4.5.2 案例二：列表渲染与交互

**需求：**显示待办事项列表，支持添加、删除、标记完成。

**Vue实现：**

```vue
<template>
  <div class="todo-app">
    <h2>待办事项</h2>

    <!-- 添加表单 -->
    <div class="add-form">
      <input
        v-model="newTodo"
        @keyup.enter="addTodo"
        placeholder="添加新任务"
      />
      <button @click="addTodo">添加</button>
    </div>

    <!-- 筛选器 -->
    <div class="filters">
      <button
        v-for="filter in filters"
        :key="filter.value"
        :class="{ active: currentFilter === filter.value }"
        @click="currentFilter = filter.value"
      >
        {{ filter.label }}
      </button>
    </div>

    <!-- 列表 -->
    <ul class="todo-list">
      <li
        v-for="todo in filteredTodos"
        :key="todo.id"
        :class="{ completed: todo.completed }"
      >
        <input type="checkbox" v-model="todo.completed" />
        <span>{{ todo.title }}</span>
        <button @click="removeTodo(todo.id)" class="delete-btn">删除</button>
      </li>
    </ul>

    <!-- 统计 -->
    <div class="stats">
      <span>总计：{{ total }}</span>
      <span>已完成：{{ completed }}</span>
      <span>待办：{{ remaining }}</span>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'

const newTodo = ref('')
const currentFilter = ref('all')

const todos = ref([
  { id: 1, title: '学习Vue', completed: true },
  { id: 2, title: '学习Avalonia', completed: false },
  { id: 3, title: '写项目', completed: false }
])

const filters = [
  { label: '全部', value: 'all' },
  { label: '待办', value: 'active' },
  { label: '已完成', value: 'completed' }
]

// 筛选后的列表
const filteredTodos = computed(() => {
  switch (currentFilter.value) {
    case 'active':
      return todos.value.filter(t => !t.completed)
    case 'completed':
      return todos.value.filter(t => t.completed)
    default:
      return todos.value
  }
})

// 统计
const total = computed(() => todos.value.length)
const completed = computed(() => todos.value.filter(t => t.completed).length)
const remaining = computed(() => todos.value.filter(t => !t.completed).length)

// 添加
function addTodo() {
  if (!newTodo.value.trim()) return

  todos.value.push({
    id: Date.now(),
    title: newTodo.value,
    completed: false
  })

  newTodo.value = ''
}

// 删除
function removeTodo(id) {
  const index = todos.value.findIndex(t => t.id === id)
  if (index > -1) {
    todos.value.splice(index, 1)
  }
}
</script>

<style scoped>

  max-width: 600px;

}

.add-form {
  display: flex;

}

  color: #666;
}
```

| 功能 | Vue | Avalonia |
