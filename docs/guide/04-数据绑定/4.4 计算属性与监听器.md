## 4.4 计算属性与监听器

### 4.4.1 computed属性迁移

**Vue的computed：**

```vue
<template>
  <div>
    <p>全名：{{ fullName }}</p>
    <p>描述：{{ description }}</p>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'

const firstName = ref('John')
const lastName = ref('Doe')
const age = ref(25)

// 只读计算属性
const fullName = computed(() =>
  `${firstName.value} ${lastName.value}`
)

// 可写计算属性
const fullNameWritable = computed({
  get() {
    return `${firstName.value} ${lastName.value}`
  },
  set(newValue) {
    [firstName.value, lastName.value] = newValue.split(' ')
  }
})

// 复杂计算属性
const description = computed(() => {
  const ageGroup = age.value >= 18 ? '成年人' : '未成年人'
  return `${fullName.value}，${ageGroup}`
})
</script>
```

**C#属性实现：**

```csharp
public class UserViewModel : ReactiveObject
{
    private string firstName = "John";
    private string lastName = "Doe";
    private int age = 25;

    public string FirstName
    {
        get => firstName;
        set
        {
            this.RaiseAndSetIfChanged(ref firstName, value);
            // 手动触发依赖属性的更新
            this.RaisePropertyChanged(nameof(FullName));
            this.RaisePropertyChanged(nameof(Description));
        }
    }

    public string LastName
    {
        get => lastName;
        set
        {
            this.RaiseAndSetIfChanged(ref lastName, value);
            this.RaisePropertyChanged(nameof(FullName));
            this.RaisePropertyChanged(nameof(Description));
        }
    }

    public int Age
    {
        get => age;
        set
        {
            this.RaiseAndSetIfChanged(ref age, value);
            this.RaisePropertyChanged(nameof(Description));
        }
    }

    // 只读计算属性（类似computed）
    public string FullName => $"{FirstName} {LastName}";

    // 复杂计算属性
    public string Description
    {
        get
        {
            string ageGroup = Age >= 18 ? "成年人" : "未成年人";
            return $"{FullName}，{ageGroup}";
        }
    }
}
```

**使用ObservableAsPropertyHelper（OAPH）：**

```csharp
using ReactiveUI;

public class UserViewModel : ReactiveObject
{
    private string firstName = "John";
    private string lastName = "Doe";

    public string FirstName
    {
        get => firstName;
        set => this.RaiseAndSetIfChanged(ref firstName, value);
    }

    public string LastName
    {
        get => lastName;
        set => this.RaiseAndSetIfChanged(ref lastName, value);
    }

    // 使用OAPH自动计算
    public readonly ObservableAsPropertyHelper<string> fullName;
    public string FullName => fullName.Value;

    public UserViewModel()
    {
        // 当FirstName或LastName改变时，自动重新计算FullName
        this.WhenAnyValue(x => x.FirstName, x => x.LastName)
            .Select(tuple =>
            {
                var (first, last) = tuple;
                return $"{first} {last}";
            })
            .ToProperty(this, x => x.FullName, out fullName);
    }
}
```

### 4.4.2 watch深度监听

**Vue的watch：**

```vue
<script setup>
import { ref, watch } from 'vue'

const count = ref(0)
const user = ref({
  name: 'Alice',
  preferences: {
    theme: 'dark'
  }
})

// 基本监听
watch(count, (newValue, oldValue) => {
  console.log(`count从${oldValue}变为${newValue}`)
})

// 深度监听对象
watch(user, (newValue) => {
  console.log('user改变了', newValue)
}, { deep: true })

// 监听特定属性
watch(() => user.value.preferences.theme, (newTheme) => {
  console.log('主题改变了', newTheme)
})

// 立即执行
watch(count, (newValue) => {
  console.log('立即执行一次', newValue)
}, { immediate: true })
</script>
```

**ReactiveUI的Subscribe：**

```csharp
using ReactiveUI;
using System.Reactive;

public class UserViewModel : ReactiveObject
{
    private int count = 0;
    private User user = new User();

    public int Count
    {
        get => count;
        set => this.RaiseAndSetIfChanged(ref count, value);
    }

    public User User
    {
        get => user;
        set => this.RaiseAndSetIfChanged(ref user, value);
    }

    public UserViewModel()
    {
        // 基本监听（类似watch）
        this.WhenAnyValue(x => x.Count)
            .Subscribe(newValue =>
            {
                Console.WriteLine($"Count变为{newValue}");
            });

        // 监听嵌套属性
        this.WhenAnyValue(x => x.User.Preferences.Theme)
            .Subscribe(newTheme =>
            {
                Console.WriteLine($"主题改变了：{newTheme}");
            });

        // 监听多个属性
        this.WhenAnyValue(x => x.User.Name, x => x.User.Age)
            .Subscribe(tuple =>
            {
                var (name, age) = tuple;
                Console.WriteLine($"用户：{name}，年龄：{age}");
            });
    }
}

public class User
{
    public string Name { get; set; } = "Alice";
    public Preferences Preferences { get; set; } = new Preferences();
}

public class Preferences
{
    public string Theme { get; set; } = "dark";
}
```

**对比Vue的deep/immediate选项：**

```csharp
// Vue: { deep: true }
// C#: 需要订阅嵌套对象的所有属性

// Vue: { immediate: true }
// C#: 使用.StartWith()
this.WhenAnyValue(x => x.Count)
    .StartWith(Count)  // 立即执行一次
    .Subscribe(value =>
    {
        Console.WriteLine($"当前值：{value}");
    });
```
